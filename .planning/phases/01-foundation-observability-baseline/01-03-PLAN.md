    ```python
    """Shared YAML config source for pydantic-settings v2.

    Pattern: each config model's model_config sets yaml_file=<path>; this source
    opens the file, yaml.safe_loads it, and returns the dict to pydantic-settings.
    """
    from __future__ import annotations

    from pathlib import Path
    from typing import Any

    import yaml
    from pydantic.fields import FieldInfo
    from pydantic_settings import BaseSettings, PydanticBaseSettingsSource


    class YamlConfigSettingsSource(PydanticBaseSettingsSource):
        """Read settings from a YAML file pinned in model_config.yaml_file."""

        def __init__(self, settings_cls: type[BaseSettings]) -> None:
            super().__init__(settings_cls)
            self._yaml_file: str | None = getattr(settings_cls.model_config, "yaml_file", None) or settings_cls.model_config.get("yaml_file")
            self._data: dict[str, Any] = {}
            if self._yaml_file:
                path = Path(self._yaml_file)
                if path.exists():
                    with path.open("r", encoding="utf-8") as f:
                        loaded = yaml.safe_load(f) or {}
                    if not isinstance(loaded, dict):
                        raise ValueError(f"Config {self._yaml_file} must deserialize to a mapping, got {type(loaded).__name__}")
                    self._data = loaded
                else:
                    raise FileNotFoundError(f"Config file not found: {self._yaml_file}")

        def get_field_value(self, field: FieldInfo, field_name: str) -> tuple[Any, str, bool]:
            value = self._data.get(field_name)
            return value, field_name, value is not None

        def __call__(self) -> dict[str, Any]:
            return self._data
    ```

    IMPORTANT — pydantic-settings override pattern (for executor):
      - `RubricConfig(_yaml_file=str(bad))` does NOT work — `_yaml_file` is not
        a BaseSettings constructor kwarg; pydantic-settings silently ignores it
        and loads the real `config/rubric.yaml` from the repo. The
        `ValidationError` assertion would never trigger — the test would pass
        against the real 5-axis config without ever exercising the validator.
      - The correct, simplest override is `monkeypatch.chdir(tmp_path)` after
        creating `tmp_path/config/rubric.yaml`. The relative `yaml_file` path
        declared in `RubricConfig.model_config` resolves against cwd, so
        chdir'ing into `tmp_path` redirects the loader to the bad fixture.
      - Alternative patterns (documented here so executor understands the
        family of working options):
          (a) monkeypatch.setattr on RubricConfig.model_config's yaml_file
              value to point at the tmp file (requires mutable copy — pydantic
              model_config is a ConfigDict, so `RubricConfig.model_config = dict(RubricConfig.model_config) | {"yaml_file": str(bad)}` works).

    Per FOUND-02: startup fails with a clear error if any field is missing/malformed.
    This CLI makes that standalone — runnable against uncommitted config drafts.
    """
    from __future__ import annotations

    import argparse
    import sys

    from pydantic import ValidationError

    from book_pipeline.cli.main import register_subcommand
    from book_pipeline.config.loader import load_all_configs


    def _add_parser(subparsers: argparse._SubParsersAction[argparse.ArgumentParser]) -> None:
        p = subparsers.add_parser(
            "validate-config",
            help="Load and validate all 4 config YAMLs + secrets from .env; exit non-zero on failure",
        )
        p.add_argument(
            "--verbose",
            "-v",
            action="store_true",
            help="Print loaded config summary (secrets always masked)",
        )
        p.set_defaults(_handler=_run)


    def _run(args: argparse.Namespace) -> int:
        try:
            cfg = load_all_configs()
        except ValidationError as exc:
            print("[FAIL] Configuration validation failed:", file=sys.stderr)
            for err in exc.errors():
                loc = " -> ".join(str(x) for x in err.get("loc", []))
                msg = err.get("msg", "validation error")
                print(f"  {loc}: {msg}", file=sys.stderr)
            return 1
        except FileNotFoundError as exc:
            print(f"[FAIL] Config file missing: {exc}", file=sys.stderr)
            return 2
        except (ValueError, OSError) as exc:
            print(f"[FAIL] Config load error: {exc}", file=sys.stderr)
            return 3

        print("[OK] All 4 configs validated successfully.")
        print(f"  voice_pin.base_model        = {cfg['voice_pin'].voice_pin.base_model}")
        print(f"  voice_pin.ft_run_id         = {cfg['voice_pin'].voice_pin.ft_run_id}")
        print(f"  rubric.rubric_version       = {cfg['rubric'].rubric_version}")
        print(f"  rubric.axes                 = {sorted(cfg['rubric'].axes.keys())}")
        print(f"  rag_retrievers              = {sorted(cfg['rag_retrievers'].retrievers.keys())}")
        print(f"  rag_retrievers.bundler_cap  = {cfg['rag_retrievers'].bundler.max_bytes} bytes")
        print(f"  mode_thresholds.regen_R     = {cfg['mode_thresholds'].mode_a.regen_budget_R}")
        print(f"  mode_thresholds.mode_b_ttl  = {cfg['mode_thresholds'].mode_b.prompt_cache_ttl}")

        secrets = cfg["secrets"]
        print("  secrets (values never printed):")
        print(f"    ANTHROPIC_API_KEY        = {'PRESENT' if secrets.is_anthropic_present() else 'MISSING'}")
        print(f"    OPENCLAW_GATEWAY_TOKEN   = {'PRESENT' if secrets.is_openclaw_present() else 'MISSING'}")
        print(f"    TELEGRAM (bot+chat_id)   = {'PRESENT' if secrets.is_telegram_present() else 'MISSING (ok for Phase 1)'}")

        return 0


    register_subcommand("validate-config", _add_parser)

import lancedb
import yaml
from pydantic import BaseModel, ConfigDict, Field, field_validator

from book_pipeline.rag.embedding import BgeM3Embedder
from book_pipeline.rag.lance_schema import open_or_create_table


_CANONICAL_QUANTITY_ID_RE = r"^[a-z0-9_]+$"


class CanonicalQuantity(BaseModel):
    model_config = ConfigDict(extra="forbid", frozen=True)
    id: str = Field(pattern=_CANONICAL_QUANTITY_ID_RE)
    name: str
    canonical_value: str
    units: str
    chapter_scope: str
    text: str = Field(min_length=1)
    source: str

    @field_validator("id")
    @classmethod
    def id_is_safe(cls, v: str) -> str:
        # Defense in depth — the Field(pattern=...) constraint already enforces this,
        # but an explicit validator makes the intent visible to readers and stable
        # against future Field-level regression. T-07-03 mitigation visible in code.
        import re
        if not re.fullmatch(_CANONICAL_QUANTITY_ID_RE, v):
            raise ValueError(
                f"CanonicalQuantity.id must match {_CANONICAL_QUANTITY_ID_RE!r}; "
                f"got {v!r} (T-07-03 mitigation)"
            )
        return v


def load_canonical_quantities_seed(yaml_path: str | Path) -> list[CanonicalQuantity]:
    with open(yaml_path, encoding="utf-8") as f:
        raw = yaml.safe_load(f) or {}
    return [CanonicalQuantity.model_validate(q) for q in raw.get("quantities", [])]


def ingest_canonical_quantities(
    *,
    db_path: Path,
    seed_yaml_path: Path,
    embedder: BgeM3Embedder,
    ingestion_run_id: str,
) -> int:
    """Ingest seed YAML rows into the 'continuity_bible' LanceDB table.

    Returns the number of rows written. Idempotent: chunk_id is deterministic
    `f"canonical:{q.id}"` and q.id is regex-validated (T-07-03).
    """
    quantities = load_canonical_quantities_seed(seed_yaml_path)
    if not quantities:
        return 0

    db = lancedb.connect(str(db_path))
    table = open_or_create_table(db, axis_name="continuity_bible")

    texts = [q.text for q in quantities]
    embeddings = embedder.embed_texts(texts)

    rows: list[dict[str, Any]] = []
    for q, emb in zip(quantities, embeddings, strict=True):
        # Safe: q.id passed `^[a-z0-9_]+$` regex via CanonicalQuantity validator.
        chunk_id = f"canonical:{q.id}"
        rows.append(
            {
                "chunk_id": chunk_id,
                "text": q.text,
                "source_file": "config/canonical_quantities_seed.yaml",
                "heading_path": f"Canonical Quantity: {q.id}",
                "rule_type": "canonical_quantity",
                "ingestion_run_id": ingestion_run_id,
                "chapter": None,  # canonical quantities are scope-aware via text, not column
                "source_chapter_sha": None,
                "embedding": emb.tolist(),
            }
        )

    # Idempotent: delete any rows with these chunk_ids first, then insert.
    # T-07-03: chunk_ids are derived from regex-validated q.id values; the
    # f-string interpolation is provably safe (alphanumeric + underscore only).
    chunk_ids = [r["chunk_id"] for r in rows]
    placeholder = ", ".join(f"'{cid}'" for cid in chunk_ids)
    try:
        table.delete(f"chunk_id IN ({placeholder})")
    except (lancedb.LanceError, ValueError) as exc:
        # Empty-table tolerance OR delete-syntax mismatch on first ingest into
        # a fresh table — either is recoverable; subsequent table.add() proceeds.
        # We intentionally do NOT swallow other exception classes (no bare except).
        import logging
        logging.getLogger(__name__).debug(
            "continuity_bible delete pre-add no-op (%s): %r", type(exc).__name__, exc
        )
    table.add(rows)
    return len(rows)


__all__ = [
    "CanonicalQuantity",
    "ingest_canonical_quantities",
    "load_canonical_quantities_seed",
]

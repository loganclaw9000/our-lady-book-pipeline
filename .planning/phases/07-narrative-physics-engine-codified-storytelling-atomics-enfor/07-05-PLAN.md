"""SceneEmbeddingCache + cosine helpers (Plan 07-05 PHYSICS-10 / D-28).

SQLite-backed embedding cache keyed by (scene_id, bge_m3_revision_sha). Pitfall 7
mitigation: cache key includes embedder revision_sha — model upgrade naturally
invalidates via missing composite-key cache hit. Pitfall 12 mitigation: db_path
is a constructor arg so tests inject tmp_path.

Pitfall 3 mitigation: BGE-M3 returns unit-normalized vectors (verified at
rag/embedding.py:104-107 normalize_embeddings=True). Cosine is dot product;
do NOT recompute norms. Add assertion |a| ≈ 1.0 on read for safety.
"""
from __future__ import annotations

import sqlite3
from datetime import UTC, datetime
from pathlib import Path
from typing import Any

import numpy as np

from book_pipeline.rag.embedding import BgeM3Embedder, EMBEDDING_DIM


class SceneEmbeddingCache:
    """SQLite-backed embedding cache.

    Production path: .planning/intel/scene_embeddings.sqlite
    Test path: tmp_path injection.
    """

    def __init__(self, db_path: Path, embedder: BgeM3Embedder) -> None:
        self.db_path = Path(db_path)
        self.embedder = embedder
        self.db_path.parent.mkdir(parents=True, exist_ok=True)
        self._conn = sqlite3.connect(str(self.db_path))
        self._conn.execute(
            "CREATE TABLE IF NOT EXISTS scene_embeddings ("
            "  scene_id TEXT NOT NULL,"
            "  bge_m3_revision_sha TEXT NOT NULL,"
            "  embedding BLOB NOT NULL,"
            "  computed_at TEXT NOT NULL,"
            "  PRIMARY KEY (scene_id, bge_m3_revision_sha)"
            ")"
        )
        self._conn.commit()

    def get_or_compute(self, scene_id: str, scene_text: str) -> np.ndarray:
        """Return embedding for scene_id; compute + persist on cache miss."""
        revision = self.embedder.revision_sha
        row = self._conn.execute(
            "SELECT embedding FROM scene_embeddings "
            "WHERE scene_id = ? AND bge_m3_revision_sha = ?",
            (scene_id, revision),
        ).fetchone()
        if row is not None:
            arr = np.frombuffer(row[0], dtype=np.float32)
            assert arr.shape == (EMBEDDING_DIM,), f"cached embedding wrong shape: {arr.shape}"
            return arr.copy()  # frombuffer returns read-only view; copy to make mutable

        arr = self.embedder.embed_texts([scene_text])[0].astype(np.float32)
        ts = datetime.now(UTC).isoformat()
        self._conn.execute(
            "INSERT OR REPLACE INTO scene_embeddings VALUES (?, ?, ?, ?)",
            (scene_id, revision, arr.tobytes(), ts),
        )
        self._conn.commit()
        return arr

    def all_prior(self, prior_scene_ids: list[str]) -> dict[str, np.ndarray]:
        """Bulk fetch cached embeddings for a list of prior scene_ids.

        Skips IDs not in cache (caller may need to populate via get_or_compute first).
        """
        revision = self.embedder.revision_sha
        out: dict[str, np.ndarray] = {}
        if not prior_scene_ids:
            return out
        placeholder = ",".join("?" * len(prior_scene_ids))
        params = (*prior_scene_ids, revision)
        rows = self._conn.execute(
            f"SELECT scene_id, embedding FROM scene_embeddings "
            f"WHERE scene_id IN ({placeholder}) AND bge_m3_revision_sha = ?",
            params,
        ).fetchall()
        for sid, blob in rows:
            arr = np.frombuffer(blob, dtype=np.float32)
            assert arr.shape == (EMBEDDING_DIM,)
            out[sid] = arr.copy()
        return out


def cosine_similarity_to_prior(
    candidate_embedding: np.ndarray,
    prior_embeddings: dict[str, np.ndarray],
) -> dict[str, float]:
    """Return {scene_id: cosine_sim}. BGE-M3 unit-norm vectors → dot product is cosine.

    Pitfall 3: trust embedder normalization; assert on read.
    """
    assert abs(float(np.linalg.norm(candidate_embedding)) - 1.0) < 1e-3, (
        "candidate embedding not unit-normalized — Pitfall 3 mitigation failed"
    )
    out: dict[str, float] = {}
    for sid, vec in prior_embeddings.items():
        assert abs(float(np.linalg.norm(vec)) - 1.0) < 1e-3, f"{sid} embedding not unit-normalized"
        out[sid] = float(np.dot(candidate_embedding, vec))
    return out


def max_cosine(candidate_embedding: np.ndarray, prior_embeddings: dict[str, np.ndarray]) -> tuple[str | None, float]:
    """Return (scene_id_of_max_match, cosine_value). Empty prior → (None, 0.0)."""
    if not prior_embeddings:
        return None, 0.0
    sims = cosine_similarity_to_prior(candidate_embedding, prior_embeddings)
    sid, sim = max(sims.items(), key=lambda kv: kv[1])
    return sid, sim


__all__ = ["SceneEmbeddingCache", "cosine_similarity_to_prior", "max_cosine"]
```

import re

class QuoteRepair(BaseModel):
    model_config = ConfigDict(extra="forbid", frozen=True)
    pattern_id: str
    line_number: int
    before: str
    after: str


# D-18 corruption pattern: ., sequences mangle dialogue ("said," → "said.,").
# Anchor line-by-line (no `.*` followed by `\s*` per Pitfall 4).
_QUOTE_COMMA_DOT_RE = re.compile(r'\.\s*,\s+')


def _normalize_quote_corruption(body: str) -> tuple[str, list[QuoteRepair]]:
    """Defensive normalizer for ., quote corruption (D-18 PHYSICS-11).

    Returns (normalized_body, repairs). On no-op, returns (body, []).
    """
    repairs: list[QuoteRepair] = []
    fixed_lines: list[str] = []
    for line_no, line in enumerate(body.splitlines(), start=1):
        if _QUOTE_COMMA_DOT_RE.search(line):
            new_line = _QUOTE_COMMA_DOT_RE.sub(", ", line)
            repairs.append(
                QuoteRepair(
                    pattern_id="dot_comma_corruption",
                    line_number=line_no,
                    before=line[:200],
                    after=new_line[:200],
                )
            )
            fixed_lines.append(new_line)
        else:
            fixed_lines.append(line)
    return "\n".join(fixed_lines), repairs
```

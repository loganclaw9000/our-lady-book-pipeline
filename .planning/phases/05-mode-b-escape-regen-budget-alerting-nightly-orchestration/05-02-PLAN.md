    ```python
    from __future__ import annotations
    from book_pipeline.interfaces.types import Event

    def _extract_axis_severity_set(event: Event) -> frozenset[tuple[str, str]]:
        """Pull (axis, severity) tuples from a role='critic' Event's extra.severities dict.
        Format: extra.severities = {'historical': 'mid', 'metaphysics': 'high', ...}."""
        severities = event.extra.get("severities", {}) if event.extra else {}
        return frozenset((axis, sev) for axis, sev in severities.items())

    def detect_oscillation(
        critic_events: list[Event],
        *,
        min_history: int = 2,
    ) -> tuple[bool, frozenset[tuple[str, str]] | None]:
        """Pure function — compare attempts N and N-2 per D-07.

        Returns (fired, repeated_tuples). repeated_tuples contains the
        (axis, severity) pairs present in BOTH N and N-2 at mid/high severity only.
        """
        if len(critic_events) < min_history:
            return False, None
        latest = _extract_axis_severity_set(critic_events[-1])
        two_back = (
            _extract_axis_severity_set(critic_events[-3])
            if len(critic_events) >= 3
            else frozenset()
        )
        common = latest & two_back
        significant = frozenset((a, s) for a, s in common if s in ("mid", "high"))
        if significant:
            return True, significant
        return False, None
    ```

    ```python
    import re
    import shutil
    from pathlib import Path
    from book_pipeline.interfaces.types import CriticResponse, SceneState, SceneStateRecord
    from book_pipeline.interfaces.scene_state_machine import transition
    from book_pipeline.interfaces.event_logger import EventLogger
    from book_pipeline.observability.event_logger import Event

    # Widened from cli/draft.py _SCENE_ID_RE (^ch(\d+)_sc(\d+)$) because
    # CriticIssue.location is free-text with embedded refs.
    _SCENE_REF_RE = re.compile(r"\bch(\d+)_sc(\d+)\b")

    def extract_implicated_scene_ids(
        response: CriticResponse,
    ) -> tuple[set[str], list[str]]:
        """Return (implicated_set, non_specific_claims). Scene IDs canonicalized
        to ch{NN:02d}_sc{II:02d}. Defensive int-cast per Plan 03-07 _parse_scene_id
        + Plan 04-04 path sanitization precedent."""
        implicated: set[str] = set()
        non_specific: list[str] = []
        for issue in response.issues:
            matches = _SCENE_REF_RE.findall(issue.location or "")
            if not matches and issue.evidence:
                matches = _SCENE_REF_RE.findall(issue.evidence)
            if matches:
                for ch_str, sc_str in matches:
                    ch, sc = int(ch_str), int(sc_str)
                    implicated.add(f"ch{ch:02d}_sc{sc:02d}")
            else:
                non_specific.append(issue.claim)
        return implicated, non_specific

    def _next_archive_rev(archive_dir: Path, scene_id: str) -> int:
        """Find next available rev{K} integer in archive/{scene_id}_rev{K}.md."""
        archive_dir.mkdir(parents=True, exist_ok=True)
        existing = list(archive_dir.glob(f"{scene_id}_rev*.md"))
        if not existing:
            return 1
        revs = [int(re.search(r"_rev(\d+)\.md$", p.name).group(1)) for p in existing]
        return max(revs) + 1

    def kick_implicated_scenes(
        implicated: set[str],
        state_dir: Path,
        drafts_dir: Path,
        event_logger: EventLogger | None,
        chapter_num: int,
        issue_refs: list[str],
    ) -> None:
        """Per A8: archive md + reset state to PENDING; emit ONE scene_kick event."""
        kicked_actual: list[str] = []
        for scene_id in sorted(implicated):
            state_path = state_dir / f"ch{chapter_num:02d}" / f"{scene_id}.state.json"
            if not state_path.exists():
                continue  # scene not in buffer; nothing to kick
            record = SceneStateRecord.model_validate_json(state_path.read_text())
            # Archive markdown if present (Pitfall 7 mitigation)
            md_path = drafts_dir / f"ch{chapter_num:02d}" / f"{scene_id}.md"
            if md_path.exists():
                archive_dir = md_path.parent / "archive"
                rev = _next_archive_rev(archive_dir, scene_id)
                shutil.move(str(md_path), str(archive_dir / f"{scene_id}_rev{rev:02d}.md"))
            # Reset state record to PENDING
            record = transition(record, SceneState.PENDING, f"scene_kick from ch{chapter_num}_fail")
            # Atomic tmp+rename (reuse _persist pattern from cli/draft.py)
            tmp = state_path.with_suffix(".tmp")
            tmp.write_text(record.model_dump_json(indent=2))
            tmp.replace(state_path)
            kicked_actual.append(scene_id)
        if event_logger is not None and kicked_actual:
            event_logger.log(Event(
                event_id=..., ts_iso=..., role="scene_kick",
                model="n/a", prompt_hash="n/a", input_tokens=0, cached_tokens=0,
                output_tokens=0, latency_ms=0, temperature=None, top_p=None,
                caller_context={"chapter_num": chapter_num},
                output_hash="n/a", mode=None,
                extra={"kicked_scenes": kicked_actual, "chapter_num": chapter_num,
                       "issue_refs": issue_refs},
            ))
    ```

    ```python
    """Cooldown LRU with TTL + atomic JSON persistence.
    Key shape: (condition: str, scope: str) → last-send-epoch-seconds float.
    """
    from __future__ import annotations
    import json
    import os
    import time
    from collections import OrderedDict
    from pathlib import Path
    from typing import Optional

    class CooldownCache:
        def __init__(
            self,
            cooldown_path: Path,
            *,
            ttl_s: int = 3600,
            now_fn=time.time,
        ) -> None:
            self.cooldown_path = Path(cooldown_path)
            self.ttl_s = ttl_s
            self._now = now_fn
            self._data: OrderedDict[tuple[str, str], float] = self._load()

        def _load(self) -> OrderedDict[tuple[str, str], float]:
            if not self.cooldown_path.exists():
                return OrderedDict()
            try:
                raw = json.loads(self.cooldown_path.read_text())
            except (json.JSONDecodeError, OSError):
                return OrderedDict()
            now = self._now()
            out: OrderedDict[tuple[str, str], float] = OrderedDict()
            for entry in raw.get("entries", []):
                c, s, t = entry["condition"], entry["scope"], entry["ts"]
                if now - t < self.ttl_s:  # prune expired on load
                    out[(c, s)] = t
            return out

        def _persist(self) -> None:
            tmp = self.cooldown_path.with_suffix(".tmp")
            self.cooldown_path.parent.mkdir(parents=True, exist_ok=True)
            tmp.write_text(json.dumps({
                "entries": [
                    {"condition": c, "scope": s, "ts": t}
                    for (c, s), t in self._data.items()
                ],
            }, indent=2))
            os.replace(str(tmp), str(self.cooldown_path))  # atomic

        def is_suppressed(self, condition: str, scope: str) -> bool:
            last = self._data.get((condition, scope))
            if last is None:
                return False
            return (self._now() - last) < self.ttl_s

        def record(self, condition: str, scope: str) -> None:
            self._data[(condition, scope)] = self._now()
            self._data.move_to_end((condition, scope))
            self._persist()
    ```

    ```python
    import subprocess
    from functools import lru_cache
    from pathlib import Path

    def _git_sha_for_chapter(chapter_num: int, repo_root: Path) -> str | None:
        """Return git SHA of latest commit touching canon/chapter_NN.md, or None."""
        path = f"canon/chapter_{chapter_num:02d}.md"
        try:
            result = subprocess.run(
                ["git", "-C", str(repo_root), "rev-list", "-1", "HEAD", "--", path],
                capture_output=True, text=True, timeout=5, check=True,
            )
            return result.stdout.strip() or None
        except (subprocess.CalledProcessError, subprocess.TimeoutExpired, FileNotFoundError):
            return None  # Pitfall 6: degrade gracefully outside git repo

    def scan_for_stale_cards(
        entity_state_result: RetrievalResult,
        repo_root: Path,
    ) -> list[ConflictReport]:
        """Compare each entity_state hit's source_chapter_sha to current canon sha."""
        stale: list[ConflictReport] = []
        # Bundle-local memoization: dict, not lru_cache (per RESEARCH.md A6 —
        # explicit per-bundle scope, no cross-request cache pollution)
        sha_by_chapter: dict[int, str | None] = {}
        for hit in entity_state_result.hits:
            card_sha = hit.metadata.get("source_chapter_sha")
            chapter = hit.metadata.get("chapter")
            if not card_sha or chapter is None:
                continue
            if chapter not in sha_by_chapter:
                sha_by_chapter[chapter] = _git_sha_for_chapter(int(chapter), repo_root)
            current_sha = sha_by_chapter[chapter]
            if current_sha is None or current_sha == card_sha:
                continue  # no git, or matching — not stale
            stale.append(ConflictReport(
                entity=hit.chunk_id,
                dimension="stale_card",
                severity="mid",
                values_by_retriever={
                    "entity_state.card_sha": card_sha,
                    "canon.head_sha": current_sha,
                },
            ))
        return stale
    ```
